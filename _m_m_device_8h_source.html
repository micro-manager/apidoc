99  class MMTime
100  {
101  long long microseconds_;
102 
103  public:
104  MMTime() : microseconds_(0LL) {}
105 
106  explicit MMTime(double uSecTotal) :
107  microseconds_(static_cast<long long>(uSecTotal))
108  {}
109 
110  explicit MMTime(long sec, long uSec) :
111  microseconds_(sec * 1'000'000LL + uSec)
112  {}
113 
114  static MMTime fromUs(long long us)
115  {
116  // Work around our lack of a constructor that directly sets the
117  // internal representation.
118  // (Note that we cannot add a constructor from 'long long' because
119  // many existing uses would then get an error (ambiguous with the
120  // 'double' overload).)
121  MMTime ret;
122  ret.microseconds_ = us;
123  return ret;
124  }
125 
126  static MMTime fromMs(double ms)
127  {
128  return MMTime(ms * 1000.0);
129  }
130 
131  static MMTime fromSeconds(long secs)
132  {
133  return MMTime(secs, 0);
134  }
135 
136  MMTime operator+(const MMTime &other) const
137  {
138  return fromUs(microseconds_ + other.microseconds_);
139  }
140 
141  MMTime operator-(const MMTime &other) const
142  {
143  return fromUs(microseconds_ - other.microseconds_);
144  }
145 
146  bool operator>(const MMTime &other) const
147  {
148  return microseconds_ > other.microseconds_;
149  }
150 
151  bool operator>=(const MMTime &other) const
152  {
153  return microseconds_ >= other.microseconds_;
154  }
155 
156  bool operator<(const MMTime &other) const
157  {
158  return microseconds_ < other.microseconds_;
159  }
160 
161  bool operator<=(const MMTime &other) const
162  {
163  return microseconds_ <= other.microseconds_;
164  }
165 
166  bool operator==(const MMTime &other) const
167  {
168  return microseconds_ == other.microseconds_;
169  }
170 
171  bool operator!=(const MMTime &other) const
172  {
173  return !(*this == other);
174  }
175 
176  double getMsec() const
177  {
178  return microseconds_ / 1000.0;
179  }
180 
181  double getUsec() const
182  {
183  return static_cast<double>(microseconds_);
184  }
185 
186  std::string toString() const {
187  long long absUs = std::abs(microseconds_);
188  long long seconds = absUs / 1'000'000LL;
189  long long fracUs = absUs - seconds * 1'000'000LL;
190  const char *sign = microseconds_ < 0 ? "-" : "";
191 
192  using namespace std;
193  ostringstream s;
194  s << sign << seconds << '.' <<
195  setfill('0') << right << setw(6) << fracUs;
196  return s.str();
197  }
198  };
