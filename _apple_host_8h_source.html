73 // Returns an iterator containing the primary (built-in) Ethernet interface. The caller is responsible for
74 // releasing the iterator after the caller is done with it.
75 static kern_return_t FindEthernetInterfaces(io_iterator_t *matchingServices)
76 {
77  kern_return_t kernResult;
78  CFMutableDictionaryRef matchingDict;
79  CFMutableDictionaryRef propertyMatchDict;
80 
81  // Ethernet interfaces are instances of class kIOEthernetInterfaceClass.
82  // IOServiceMatching is a convenience function to create a dictionary with the key kIOProviderClassKey and
83  // the specified value.
84  matchingDict = IOServiceMatching(kIOEthernetInterfaceClass);
85 
86  // Note that another option here would be:
87  // matchingDict = IOBSDMatching("en0");
88 
89  if (NULL == matchingDict) {
90  printf("IOServiceMatching returned a NULL dictionary.\n");
91  }
92  else {
93  // Each IONetworkInterface object has a Boolean property with the key kIOPrimaryInterface. Only the
94  // primary (built-in) interface has this property set to TRUE.
95 
96  // IOServiceGetMatchingServices uses the default matching criteria defined by IOService. This considers
97  // only the following properties plus any family-specific matching in this order of precedence
98  // (see IOService::passiveMatch):
99  //
100  // kIOProviderClassKey (IOServiceMatching)
101  // kIONameMatchKey (IOServiceNameMatching)
102  // kIOPropertyMatchKey
103  // kIOPathMatchKey
104  // kIOMatchedServiceCountKey
105  // family-specific matching
106  // kIOBSDNameKey (IOBSDNameMatching)
107  // kIOLocationMatchKey
108 
109  // The IONetworkingFamily does not define any family-specific matching. This means that in
110  // order to have IOServiceGetMatchingServices consider the kIOPrimaryInterface property, we must
111  // add that property to a separate dictionary and then add that to our matching dictionary
112  // specifying kIOPropertyMatchKey.
113 
114  propertyMatchDict = CFDictionaryCreateMutable(kCFAllocatorDefault, 0,
115  &kCFTypeDictionaryKeyCallBacks,
116  &kCFTypeDictionaryValueCallBacks);
117 
118  if (NULL == propertyMatchDict) {
119  printf("CFDictionaryCreateMutable returned a NULL dictionary.\n");
120  }
121  else {
122  // Set the value in the dictionary of the property with the given key, or add the key
123  // to the dictionary if it doesn't exist. This call retains the value object passed in.
124  CFDictionarySetValue(propertyMatchDict, CFSTR(kIOPrimaryInterface), kCFBooleanTrue);
125 
126  // Now add the dictionary containing the matching value for kIOPrimaryInterface to our main
127  // matching dictionary. This call will retain propertyMatchDict, so we can release our reference
128  // on propertyMatchDict after adding it to matchingDict.
129  CFDictionarySetValue(matchingDict, CFSTR(kIOPropertyMatchKey), propertyMatchDict);
130  CFRelease(propertyMatchDict);
131  }
132  }
133 
134  // IOServiceGetMatchingServices retains the returned iterator, so release the iterator when we're done with it.
135  // IOServiceGetMatchingServices also consumes a reference on the matching dictionary so we don't need to release
136  // the dictionary explicitly.
137  kernResult = IOServiceGetMatchingServices(kIOMasterPortDefault, matchingDict, matchingServices);
138  if (KERN_SUCCESS != kernResult) {
139  printf("IOServiceGetMatchingServices returned 0x%08x\n", kernResult);
140  }
141 
142  return kernResult;
143 }

145 // Given an iterator across a set of Ethernet interfaces, return the MAC address of the last one.
146 // If no interfaces are found the MAC address is set to an empty string.
147 // In this sample the iterator should contain just the primary interface.
148 static kern_return_t GetMACAddress(io_iterator_t intfIterator, UInt8 *MACAddress, UInt8 bufferSize)
149 {
150  kern_return_t kernResult = KERN_FAILURE;
151 
152  // Make sure the caller provided enough buffer space. Protect against buffer overflow problems.
153  if (bufferSize < kIOEthernetAddressSize) {
154  return kernResult;
155  }
156 
157  // Initialize the returned address
158  bzero(MACAddress, bufferSize);
159 
160  // IOIteratorNext retains the returned object, so release it when we're done with it.
161  while (io_object_t intfService = IOIteratorNext(intfIterator))
162  {
163  io_object_t controllerService;
164  CFTypeRef MACAddressAsCFData;
165 
166  // IONetworkControllers can't be found directly by the IOServiceGetMatchingServices call,
167  // since they are hardware nubs and do not participate in driver matching. In other words,
168  // registerService() is never called on them. So we've found the IONetworkInterface and will
169  // get its parent controller by asking for it specifically.
170 
171  // IORegistryEntryGetParentEntry retains the returned object, so release it when we're done with it.
172  kernResult = IORegistryEntryGetParentEntry(intfService,
173  kIOServicePlane,
174  &controllerService);
175 
176  if (KERN_SUCCESS != kernResult) {
177  printf("IORegistryEntryGetParentEntry returned 0x%08x\n", kernResult);
178  }
179  else {
180  // Retrieve the MAC address property from the I/O Registry in the form of a CFData
181  MACAddressAsCFData = IORegistryEntryCreateCFProperty(controllerService,
182  CFSTR(kIOMACAddress),
183  kCFAllocatorDefault,
184  0);
185  if (MACAddressAsCFData) {
186 
187  // Get the raw bytes of the MAC address from the CFData
188  CFDataGetBytes((const __CFData*)MACAddressAsCFData, CFRangeMake(0, kIOEthernetAddressSize), MACAddress);
189  CFRelease(MACAddressAsCFData);
190  }
191 
192  // Done with the parent Ethernet controller object so we release it.
193  (void) IOObjectRelease(controllerService);
194  }
195 
196  // Done with the Ethernet interface object so we release it.
197  (void) IOObjectRelease(intfService);
198  }
199 
200  return kernResult;
201 }
